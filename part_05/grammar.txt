program
  : expr+

expr
  : term
  | call

call
  : identifier '(' expr* (, expr)* ')'

assignment
  : 'def' identifier '=' expr

binary_operation
  : atom OPERATOR atom

atom
  : number
  | '(' expr ')'
  | identifier
  | MINUS* atom

identifier
  : starting_char (id_char)*

number
  : digit+ (".")? digit*

id_char
  : (letter | number | _)

starting_char
  : (letter | _)

digit
  : [0-9]

letter
  : [a-zA-Z]

OPERATOR
  : +
  | -
  | *
  | /
  | %
  | =
